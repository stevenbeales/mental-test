def dump_and_load(obj, trace=false)
  json = Oj.dump(obj, :indent => 2)
  puts json if trace
  loaded = Oj.strict_load(json);
  if obj.nil?
    assert_nil(loaded)
  else
    assert_equal(obj, loaded)
  end
  loaded
end

def test_bigdecimal_load
  orig = BigDecimal('80.51')
  json = Oj.dump(orig, :mode => :strict, :bigdecimal_as_decimal => true)
  bg = Oj.load(json, :mode => :strict, :bigdecimal_load => true)
  assert_equal(BigDecimal, bg.class)
  assert_equal(orig, bg)
end

def test_encode
  opts = Oj.default_options
  Oj.default_options = { :ascii_only => false }
  unless 'jruby' == $ruby
    dump_and_load("ぴーたー", false)
  end
  Oj.default_options = { :ascii_only => true }
  json = Oj.dump("ぴーたー")
  assert_equal(%Q{"\\u3074\\u30fc\\u305f\\u30fc"}, json)
  unless 'jruby' == $ruby
    dump_and_load("ぴーたー", false)
  end
  Oj.default_options = opts
end

  def test_io_file
    filename = File.join(File.dirname(__FILE__), 'open_file_test.json')
    File.open(filename, 'w') { |f| f.write(%Q{{
