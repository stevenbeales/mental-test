    This cop checks for whitespace within string interpolations.

def autocorrect(node)
  new_source = style == :no_space ? node.source : " #{node.source} "
  lambda do |corrector|
    corrector.replace(
      range_with_surrounding_space(range: node.source_range),
      new_source
    )
  end
end
