def autocorrect(node)
  lambda do |corrector|
    corrector.replace(node.source_range, node.source.delete(%q('")))
  end
end

def on_sym(node)
  return unless node.source =~ /\A:["'][A-Za-z_]\w*["']\z/

  add_offense(node)
end
