def <=>(other)
  COMPARISON_ATTRIBUTES.each do |attribute|
    result = send(attribute) <=> other.send(attribute)
    return result unless result.zero?
  end
  0
end

def column_length
  if first_line == last_line
    column_range.count
  else
    source_line.length - column
  end
end
