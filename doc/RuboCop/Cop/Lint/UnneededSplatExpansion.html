    class RuboCop::Cop::Lint::UnneededSplatExpansion

    This cop checks for unneeded usages of splat expansion

    @example

    # bad

a = *[1, 2, 3]
a = *'a'
a = *1

begin
  foo
rescue *[StandardError, ApplicationError]
  bar
end

case foo
when *[1, 2, 3]
  bar
else
  baz
end

    @example

    # good

c = [1, 2, 3]
a = *c
a, b = *c
a, *b = *c
a = *1..10
a = ['a']

begin
  foo
rescue StandardError, ApplicationError
  bar
end

case foo
when *[1, 2, 3]
  bar
else
  baz
end

def autocorrect(node)
  range, content = replacement_range_and_content(node)

  lambda do |corrector|
    corrector.replace(range, content)
  end
end

def on_splat(node)
  unneeded_splat_expansion(node) do
    if array_splat?(node) &&
       (method_argument?(node) || part_of_an_array?(node))
      add_offense(node, message: ARRAY_PARAM_MSG)
    else
      add_offense(node)
    end
  end
end
