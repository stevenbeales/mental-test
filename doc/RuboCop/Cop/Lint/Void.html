    RuboCop::Cop::Cop

    class RuboCop::Cop::Lint::Void

    This cop checks for operators, variables and literals used in void context.

    @example

    # bad

def some_method
  some_num * 10
  do_something
end

    @example

    # bad

def some_method(some_var)
  some_var
  do_something
end

    @example

    # good

def some_method
  do_something
  some_num * 10
end

    @example

    # good

def some_method(some_var)
  do_something
  some_var
end
