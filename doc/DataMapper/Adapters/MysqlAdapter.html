def disable_referential_integrity
  old = select("SELECT @@FOREIGN_KEY_CHECKS;")
  begin
    execute("SET FOREIGN_KEY_CHECKS = 0;")
    yield
  ensure
    execute("SET FOREIGN_KEY_CHECKS = ?", *old)
  end
end
