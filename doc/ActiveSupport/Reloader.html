def self.reload!
  executor.wrap do
    new.tap do |instance|
      begin
        instance.run!
      ensure
        instance.complete!
      end
    end
  end
  prepare!
end

def release_unload_lock!
  if @locked
    @locked = false
    ActiveSupport::Dependencies.interlock.done_unloading
  end
end

def require_unload_lock!
  unless @locked
    ActiveSupport::Dependencies.interlock.start_unloading
    @locked = true
  end
end
