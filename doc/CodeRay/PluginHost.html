    module CodeRay::PluginHost

    PluginHost¶ ↑

    A simple subclass/subfolder plugin system.

    Example:

    class Generators
  extend PluginHost
  plugin_path 'app/generators'
end

class Generator
  extend Plugin
  PLUGIN_HOST = Generators
end

class FancyGenerator < Generator
  register_for :fancy
end

Generators[:fancy]  #-> FancyGenerator
# or
CodeRay.require_plugin 'Generators/fancy'
# or
Generators::Fancy

def default id = nil
  if id
    id = validate_id id
    raise "The default plugin can't be named \"default\"." if id == :default
    plugin_hash[:default] = id
  else
    load :default
  end
end

def list
  Dir[path_to('*')].select do |file|

  end

    Map a plugin_id to another.

    Usage: Put this in a file plugin_path/_map.rb.

    class MyColorHost < PluginHost
  map :navy => :dark_blue,
    :maroon => :brown,
    :luna => :moon

    # File vendor/bundle/ruby/2.4.0/gems/coderay-1.1.2/lib/coderay/helpers/plugin_host.rb, line 95
def map hash
  for from, to in hash
    from = validate_id from
    to   = validate_id to
    plugin_hash[from] = to unless plugin_hash.has_key? from
  end
end

    make_plugin_hash() click to toggle source

    Return a plugin hash that automatically loads plugins.

    # File vendor/bundle/ruby/2.4.0/gems/coderay-1.1.2/lib/coderay/helpers/plugin_host.rb, line 172
def make_plugin_hash
  Hash.new do |h, plugin_id|
    id = validate_id(plugin_id)
    path = path_to id
    begin
      require path
    rescue LoadError => boom
      if h.has_key?(:default)
        h[:default]
      else
        raise PluginNotFound, '%p could not load plugin %p: %s' % [self, id, boom]
      end
    else
      # Plugin should have registered by now
      if h.has_key? id
        h[id]
      else
        raise PluginNotFound, "No #{self.name} plugin for #{id.inspect} found in #{path}."
      end
    end
  end
end

    validate_id(id) click to toggle source

    Converts id to a valid plugin ID String, or returns nil.

    Raises ArgumentError for all other objects, or if the given String includes non-alphanumeric characters (W).

    # File vendor/bundle/ruby/2.4.0/gems/coderay-1.1.2/lib/coderay/helpers/plugin_host.rb, line 204
def validate_id id
  case id
  when Symbol
    id.to_s
  when String
    if id[/\w+/] == id
      id.downcase
    else
      raise ArgumentError, "Invalid id given: #{id}"
    end
  else
    raise ArgumentError, "Symbol or String expected, but #{id.class} given."
  end
end
