def initialize(name = :guard, options = "")
  @name = name
  @options = options

  yield self if block_given?

  desc "Starts Guard with options: '#{options}'"
  task name => ["#{name}:start"]

  namespace(name) do
    desc "Starts Guard with options: '#{options}'"
    task(:start) do
      ::Guard::CLI.start(options.split)
    end
  end
end
