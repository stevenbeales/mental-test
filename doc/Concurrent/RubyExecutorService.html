def kill
  synchronize do
    break if shutdown?
    self.ns_auto_terminate = false
    stop_event.set
    ns_kill_execution
    stopped_event.set
  end
  true
end

def shutdown
  synchronize do
    break unless running?
    self.ns_auto_terminate = false
    stop_event.set
    ns_shutdown_execution
  end
  true
end
