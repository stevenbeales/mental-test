def ensure_integer(value)
  unless value.is_a?(Integer)
    raise ArgumentError.new("#{value} is not an Integer")
  end
  value
end

def ensure_lower_bound(value)
  if value < MIN_VALUE
    raise RangeError.new("#{value} is less than the maximum value of #{MIN_VALUE}")
  end
  value
end
