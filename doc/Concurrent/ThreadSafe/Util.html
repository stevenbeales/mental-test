      def self.make_synchronized_on_rbx(klass)
        klass.class_eval do
          private
          def _mon_initialize
            @_monitor = Monitor.new unless @_monitor # avoid double initialisation
          end

          def self.allocate
            obj = super
            obj.send(:_mon_initialize)
            obj
          end
        end
