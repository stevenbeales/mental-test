def close_pipes
  read.close unless read.closed?
  write.close unless write.closed?
end

def work(data)
  begin
    Marshal.dump(data, write)
  rescue Errno::EPIPE
    raise DeadWorker
  end

  result = begin
    Marshal.load(read)
  rescue EOFError
    raise DeadWorker
  end
  raise result.exception if ExceptionWrapper === result
  result
end
