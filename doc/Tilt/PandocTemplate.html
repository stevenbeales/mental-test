def pandoc_options
  options.reduce([]) do |sum, (k,v)|
    case v
    when true
      sum << (tilt_to_pandoc_mapping[k] || k)
    when false
      sum
    else
      sum << { k => v }
    end
  end
end

def tilt_to_pandoc_mapping
  { :smartypants => :smart,
    :escape_html => { :f => 'markdown-raw_html' },
    :commonmark => { :f => 'commonmark' },
    :markdown_strict => { :f => 'markdown_strict' }
  }
end
