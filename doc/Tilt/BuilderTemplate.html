def evaluate(scope, locals, &block)
  xml = (locals[:xml] || ::Builder::XmlMarkup.new(options))

  if data.respond_to?(:to_str)
    if !locals[:xml]
      locals = locals.merge(:xml => xml)
    end
    return super(scope, locals, &block)
  end

  data.call(xml)
  xml.target!
end
