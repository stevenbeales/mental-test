def evaluate(scope, locals)
  if data.respond_to?(:to_str)
    wrapper = proc { yield.sub(DOCUMENT_HEADER, "") } if block_given?
    super(scope, locals, &wrapper)
  else
    ::Nokogiri::XML::Builder.new.tap(&data).to_xml
  end
end
