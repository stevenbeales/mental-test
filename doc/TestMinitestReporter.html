def error_test

  @et

def new_composite_reporter
  reporter = Minitest::CompositeReporter.new
  reporter << Minitest::SummaryReporter.new(self.io)
  reporter << Minitest::ProgressReporter.new(self.io)

  def reporter.first
    reporters.first
  end

  def reporter.results
    first.results
  end

  def reporter.count
    first.count
  end

  def reporter.assertions
    first.assertions
  end

  reporter
end
