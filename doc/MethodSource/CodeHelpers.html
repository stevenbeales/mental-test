    # File vendor/bundle/ruby/2.4.0/gems/method_source-0.9.0/lib/method_source/code_helpers.rb, line 66
def complete_expression?(str)
  old_verbose = $VERBOSE
  $VERBOSE = nil

  catch(:valid) do
    eval("BEGIN{throw :valid}\n#{str}")
  end

  # Assert that a line which ends with a , or \ is incomplete.
  str !~ /[,\]\s*\z/
rescue IncompleteExpression
  false
ensure
  $VERBOSE = old_verbose
end

def expression_at(file, line_number, options={})
  options = {
    :strict  => false,
    :consume => 0
  }.merge!(options)

  lines = file.is_a?(Array) ? file : file.each_line.to_a

  relevant_lines = lines[(line_number - 1)..-1] || []

  extract_first_expression(relevant_lines, options[:consume])
rescue SyntaxError => e
  raise if options[:strict]

  begin
    extract_first_expression(relevant_lines) do |code|
      code.gsub(/\#\{.*?\}/, "temp")
    end
  rescue SyntaxError
    raise e
  end
end
