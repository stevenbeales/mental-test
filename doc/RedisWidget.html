def save
  unless connection.get(self.class.to_s + ':id')
    @id = 0
    connection.set(self.class.to_s + ':id', @id)
  end
  @id = connection.incr(self.class.to_s + ':id')
  connection.set(self.class.to_s + ':%d:name' % @id, @name)
end
