def assert_triggered expected, klass = Minitest::Assertion
  @assertion_count += 1

  e = assert_raises klass do
    yield
  end

  msg = e.message.sub(/(---Backtrace---).*/m, "\1")
  msg.gsub!(/\(oid=[-0-9]+\)/, "(oid=N)")

  assert_equal expected, msg
end
