def between other
  if equals_quoted?(other.begin, -Float::INFINITY)
    if equals_quoted?(other.end, Float::INFINITY)
      not_in([])
    elsif other.exclude_end?
      lt(other.end)
    else
      lteq(other.end)
    end
  elsif equals_quoted?(other.end, Float::INFINITY)
    gteq(other.begin)
  elsif other.exclude_end?
    gteq(other.begin).and(lt(other.end))
  else
    left = quoted_node(other.begin)
    right = quoted_node(other.end)
    Nodes::Between.new(self, left.and(right))
  end
end

    def in other
      case other
      when Arel::SelectManager
        Arel::Nodes::In.new(self, other.ast)
      when Range
        if $VERBOSE
          warn <<-eowarn
Passing a range to `#in` is deprecated. Call `#between`, instead.
          eowarn
        end
        between(other)
      when Enumerable
        Nodes::In.new self, quoted_array(other)
      else
        Nodes::In.new self, quoted_node(other)
      end
    end

def not_between other
  if equals_quoted?(other.begin, -Float::INFINITY)
    if equals_quoted?(other.end, Float::INFINITY)
      self.in([])
    elsif other.exclude_end?
      gteq(other.end)
    else
      gt(other.end)
    end
  elsif equals_quoted?(other.end, Float::INFINITY)
    lt(other.begin)
  else
    left = lt(other.begin)
    right = if other.exclude_end?
      gteq(other.end)
    else
      gt(other.end)
    end
    left.or(right)
  end
end
