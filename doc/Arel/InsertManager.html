def insert fields
  return if fields.empty?

  if String === fields
    @ast.values = Nodes::SqlLiteral.new(fields)
  else
    @ast.relation ||= fields.first.first.relation

    values = []

    fields.each do |column, value|
      @ast.columns << column
      values << value
    end
    @ast.values = create_values values, @ast.columns
  end
end
