def to_config(name)
  buf = []
  buf << "rbx.platform.#{name}.#{@name}.offset = #{@offset}"
  buf << "rbx.platform.#{name}.#{@name}.size = #{@size}"
  buf << "rbx.platform.#{name}.#{@name}.type = #{@type}" if @type
  buf
end
