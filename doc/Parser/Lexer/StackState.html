def lexpop
  @stack = ((@stack >> 1) | (@stack & 1))
  @stack[0] == 1
end

def pop
  bit_value = @stack & 1
  @stack  >>= 1

  bit_value == 1
end

def push(bit)
  bit_value = bit ? 1 : 0
  @stack = (@stack << 1) | bit_value

  bit
end
