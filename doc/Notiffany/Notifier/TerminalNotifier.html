def _perform_notify(message, opts = {})
  title = [opts[:app_name], opts[:type].downcase.capitalize].join(" ")
  opts = {
    title: title
  }.merge(opts)
  opts[:message] = message
  opts[:title] ||= title
  opts.delete(:image)
  opts.delete(:app_name)

  ::TerminalNotifier::Guard.execute(false, opts)
end
