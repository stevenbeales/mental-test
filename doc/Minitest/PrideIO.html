    pride(string) click to toggle source

    Color a string.

    # File vendor/bundle/ruby/2.4.0/gems/minitest-5.11.3/lib/minitest/pride_plugin.rb, line 89
def pride string
  string = "*" if string == "."
  c = @colors[@index % @size]
  @index += 1
  "#{ESC}#{c}m#{string}#{NND}"
end

    Wrap print to colorize the output.

    # File vendor/bundle/ruby/2.4.0/gems/minitest-5.11.3/lib/minitest/pride_plugin.rb, line 60
def print o
  case o
  when "." then
    io.print pride o
  when "E", "F" then
    io.print "#{ESC}41m#{ESC}37m#{o}#{NND}"
  when "S" then
    io.print pride o
  else
    io.print o
  end
end
