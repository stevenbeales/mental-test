def run(err, out)
  begin
    DRb.start_service("druby://localhost:0")
  rescue SocketError, Errno::EADDRNOTAVAIL
    DRb.start_service("druby://:0")
  end
  spec_server = DRbObject.new_with_uri("druby://127.0.0.1:#{drb_port}")
  spec_server.run(drb_argv, err, out)
end
