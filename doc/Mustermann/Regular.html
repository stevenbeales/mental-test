def initialize(string, check_anchors: true, **options)
  string = $1 if string.to_s =~ /\A\(\?\-mix\:(.*)\)\Z/ && string.inspect == "/#$1/"
  string = string.source.gsub!(/(?<!\)(?:\s|#.*$)/, '') if extended_regexp?(string)
  @check_anchors = check_anchors
  super(string, **options)
end
