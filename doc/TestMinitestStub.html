def test_dynamic_method
  @assertion_count = 2

  dynamic = Class.new do
    def self.respond_to? meth
      meth == :found
    end

    def self.method_missing meth, *args, &block
      if meth == :found
        false
      else
        super
      end
    end
  end

  val = dynamic.stub(:found, true) do |s|
    s.found
  end

  @tc.assert_equal true, val
  @tc.assert_equal false, dynamic.found
end
