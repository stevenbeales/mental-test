    class Pry::Pager::SystemPager

    `SystemPager` buffers output until we're pretty sure it's at least a page long, then invokes an external pager and starts streaming output to it. If `#close` is called before then, it just prints out the buffered content.

def self.available?
  if @system_pager.nil?
    @system_pager = begin
      pager_executable = default_pager.split(' ').first
      if Pry::Helpers::BaseHelpers.windows? || Pry::Helpers::BaseHelpers.windows_ansi?
        %x`where #{pager_executable}`
      else
        %x`which #{pager_executable}`
      end
      $?.success?
    rescue
      false
    end
  else
    @system_pager
  end
end

def self.default_pager
  pager = ENV["PAGER"] || ""

  # Default to less, and make sure less is being passed the correct options
  if pager.strip.empty? or pager =~ /^less\b/
    pager = "less -R -F -X"
  end

  pager
end

def write(str)
  if invoked_pager?
    write_to_pager str
  else
    @tracker.record str
    @buffer << str

    if @tracker.page?
      write_to_pager @buffer
    end
  end
rescue Errno::EPIPE
  raise StopPaging
end
