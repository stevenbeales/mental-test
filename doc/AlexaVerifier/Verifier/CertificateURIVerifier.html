def valid!(uri)
  begin
    uri = URI.parse(uri)
  rescue URI::InvalidURIError => e
    puts e

    raise AlexaVerifier::InvalidCertificateURIError,
          "#{uri} : #{e.message}"
  end

  test_validations(uri)

  test_path(uri)

  true
end

def valid?(uri)
  begin
    valid!(uri)
  rescue AlexaVerifier::InvalidCertificateURIError => e
    puts e

    return false
  end

  true
end
