    Extract the name of the missing constant from the exception message.

    begin
  HelloWorld
rescue NameError => e
  e.missing_name
end
# => "HelloWorld"

    # File vendor/bundle/ruby/2.4.0/gems/activesupport-5.1.5/lib/active_support/core_ext/name_error.rb, line 10
def missing_name
  if /undefined local variable or method/ !~ message
    $1 if /((::)?([A-Z]\w*)(::[A-Z]\w*)*)$/ =~ message
  end
end

def missing_name?(name)
  if name.is_a? Symbol
    self.name == name
  else
    missing_name == name.to_s
  end
end
