def self.usable?
  return false unless super
  require 'wdm'
  true
rescue LoadError
  _log :debug, format('wdm - load failed: %s:%s', $ERROR_INFO,
                      $ERROR_POSITION * "\n")

  Kernel.warn BUNDLER_DECLARE_GEM
  false
end
