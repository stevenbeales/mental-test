def divide_seconds(seconds)
  hours, seconds   = seconds.divmod(3600)
  minutes, seconds = seconds.divmod(60)

  [hours, minutes, seconds]
end

def start
  self.started_at = stopped? ? time.now - (stopped_at - started_at) : time.now
  self.stopped_at = nil
end
