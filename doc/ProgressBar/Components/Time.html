def out_of_bounds_time_format=(format)
  unless OOB_TIME_FORMATS.include? format
    fail 'Invalid Out Of Bounds time format.  Valid formats are ' +
      OOB_TIME_FORMATS.inspect
  end

  @out_of_bounds_time_format = format
end
