16# Represents a psychometric item
  
17# Similar to a question with metadata
  
18class Item < ApplicationRecord
  
19  belongs_to :instrument, inverse_of: :items
  
20  belongs_to :response_scale, inverse_of: :items, optional: true
  
21  has_many :choices, through: :response_scale
  
22  validates :name, presence: true
  
23  validates :instrument, presence: true
  
24  validates_uniqueness_of :name
  
25  validates_length_of :name, \
  
26                      within: 2..20, \
  
27                      too_long: 'pick a shorter name', \
  
28                      too_short: 'pick a longer name'
  
29
  
30  def find_choice_by_value(value)
  
31    choices.find_by_response_scale_id_and_value(response_scale.id, value)
  
32  end
