1# frozen_string_literal: true
  
2
  
3require './models/init'
  
4
  
5RSpec.describe Visit, type: :model do
  
6  subject { described_class.find_or_create_by! survey: survey, name: AppConstants::TEST_VISIT, user: user }
  
7  let!(:survey) { TestFactory.test_survey }
  
8  let!(:user) { TestFactory.test_user }
  
 9  let!(:assessment) { Assessment.find_or_create_by! visit: subject }
  
10
  
11  describe '.create!' do
  
12    context 'no survey and user' do
  
13      it { expect { described_class.create! }.to raise_error ActiveRecord::RecordInvalid }
  
14    end
  
15
  
16    context 'no user' do
  
17      it { expect { described_class.create! survey: survey }.to raise_error ActiveRecord::RecordInvalid }
  
18    end
  
19
  
20    context 'no survey' do
  
21      it { expect { described_class.create! user: user }.to raise_error ActiveRecord::RecordInvalid }
  
22    end
  
23
  
24    context 'with user and survey' do
  
25      it { expect { described_class.find_or_create_by! user: user, survey: survey }.not_to raise_error }
  
26    end
  
27  end
  
28
  
29  describe '#to_s' do
  
30    it do
  
31      ur = User.find_or_create_by! username: 'bernie'
  
32      v = described_class.find_or_create_by! user: ur, number: 1, survey: survey
  
33      expect(v.to_s).to eq("bernie #{survey.name} #{v.number}")
  
34    end
  
35  end
  
36
  
37  describe '#eq' do
  
38    context 'same user, survey and name' do
  
39      ur = User.find_or_create_by! username: 'user compare'
  
40      sy = Survey.find_or_create_by! name: 'survey compare'
  
41      v1 = described_class.create! user: ur, name: 'visit 1', survey: sy
  
42      v2 = described_class.where(user_id: v1.user.id, name: v1.name, survey_id: v1.survey.id).first
  
43      it { expect(v1.id).to eq(v2.id) }
  
44      ur.destroy!
  
45      sy.destroy!
  
46    end
  
47  end
  
48
  
49  describe '#destroy!' do
  
50    it 'destroys assessment' do
  
51      cached_id = assessment.id
  
52      v1 = described_class.find_or_create_by! user: user, name: 'visit 1', survey: survey, number: 2
  
53      v1.assessments.concat(assessment)
  
54      v1.destroy
  
55      expect { Assessment.find(cached_id) }.to raise_error(ActiveRecord::RecordNotFound)
  
56    end
  
57
  
58    it 'destroys assessments' do
  
59      subject.assessments.each(&:destroy!)
  
60      expect(subject.assessments.count.to_s).to eq '0'
  
61    end
  
62  end
  
63
  
64  describe '#concat assessments' do
  
65    it do
  
66      ass1 = Assessment.create! visit: subject, order_number: 24
  
67      ass2 = Assessment.create! visit: subject, order_number: 2
  
68      subject.assessments.concat [ass1, ass2]
  
69      expect(subject.assessments.count.to_s).to eq '3'
  
70      subject.assessments.each(&:destroy!)
  
71    end
  
72  end
  
73
  
74  describe '#created_at today' do
  
75    # expect record to be created within the last 5 minutes to check timestamp works
  
76    it { expect(Time.now - subject.created_at).to be < 300 }
  
77  end
  
78end
