 9  let!(:ur) { TestFactory.test_user }

18  describe '.create!' do
  
19    context 'without assessment or name' do
  
20      it { expect { described_class.create! }.to raise_error ActiveRecord::RecordInvalid }
  
21    end
  
22
  
23    context 'without assessment' do
  
24      it { expect { described_class.create!(name: 'total') }.to raise_error ActiveRecord::RecordInvalid }
  
25    end

27    context 'without name' do
  
28      it { expect { described_class.create!(assessment: ass) }.to raise_error ActiveRecord::RecordInvalid }
  
29    end

31    context 'assessment and name' do
  
32      it { expect { Score.create_with(score: 8).find_or_create_by!(assessment: ass, name: 'tote') }.not_to raise_error }
  
33    end
  
34
  
35    context 'duplicate assessment and name' do
  
36      it do
  
37        expect do
  
38          described_class.create!(assessment: ass, name: 'tote', order_number: 1).to raise_error ActiveRecord::RecordInvalid
  
39        end
  
40      end
  
41
  
42      it { expect { described_class.create!(assessment: ass, name: 'tote', order_number: 2).not_to raise_error } }
  
43    end
  
44  end
  
45
  
46  describe '#destroy!' do
  
47    context 'preserves assessment' do
  
48      it do

53  end

55  describe 'destroys multiple scores' do

64      ass.scores.concat([rep2, rep3])
  
65      expect(subject.assessment.scores.count.to_s).to eq '3'

69  end
