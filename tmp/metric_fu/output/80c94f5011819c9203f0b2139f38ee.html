3require './models/init'
  
4
  
5RSpec.describe User, type: :model do
  
6  subject { TestFactory.test_user }
  
7
  
8  describe '.authenticate' do
  
 9    it 'creates a user if one does not exist' do
  
10      expect { described_class.authenticate(Faker::Internet.unique.user_name(5..20)) }.to change { User.count }.by(1)
  
11    end
  
12
  
13    it 'retrieves a user if name and access token exists' do
  
14      timmy = create(:timmy)
  
15      expect { described_class.authenticate(timmy.username) }.not_to(change { described_class.count })
  
16      expect(described_class.authenticate(timmy.username).username).to eq timmy.username
  
17      expect(described_class.authenticate(timmy.username).access_token).to eq timmy.access_token
  
18    end
  
19
  
20    it 'does not create a user without a username' do
  
21      expect { described_class.authenticate('') }.to change { described_class.count }.by(0)
  
22    end
  
23
  
24    it 'does not create a user with a name shorter than 5 characters' do
  
25      expect { described_class.authenticate('Lisa') }.to change { described_class.count }.by(0)
  
26    end
  
27  end
  
28
  
29  describe 'Saving to a database' do
  
30    it 'starts out unpersisted' do
  
31      user = described_class.new
  
32      expect(user.id).to be_nil
  
33    end
  
34
  
35    it 'can be persisted' do
  
36      user = build(:timmy)
  
37      user.save
  
38
  
39      persisted_user = described_class.find_by_username 'Timmy'
  
40      expect(persisted_user.id).not_to be_nil
  
41      expect(persisted_user.username).to eq 'Timmy'
  
42      expect(persisted_user.access_token).to eq 'AccessToken'
  
43    end
  
44
  
45    it 'has unique name' do
  
46      expect { described_class.authenticate('Timmy') }.to change { described_class.count }.by(0)
  
47    end
  
48  end
  
49
  
50  describe '#to_s' do
  
51    it do
  
52      user = build(:timmy)
  
53      expect(user.to_s).to eq(user.username)
  
54    end
  
55  end
  
56
  
57  describe '#created_at today' do
  
58    # expect record to be created within the last 5 minutes to check timestamp works
  
59    it { expect(Time.now - subject.created_at).to be < 300 }
  
60  end
  
61end
